Для кого бокс?

Бокс = по одному товару из каждой детской подкатегории (в зависимости от пола) + упаковка (наполнитель, лента, коробка) — детерминированно по номеру заказа, один и тот же номер всегда даёт один и тот же набор, пока хватает остатков.

Подключение к Google Диску и Таблице

URL сохраняется в этом браузере. Пока подключение не настроено, все данные (в т.ч. тестовые товары ниже) хранятся только локально — можно сразу играться с приложением.

Как это подключить по-настоящему

Я не могу зайти в ваш Google Диск отсюда — картинки и структуру папок нужно связать через маленький скрипт Google Apps Script, который стоит бесплатно и работает прямо внутри самой Google Таблицы «Data base». Он делает две вещи: 1) отдаёт/сохраняет строки таблицы (товары, магазины, историю) — это уже поддержано полем выше; 2) по ID товара ищет файл ID.png / ID.jpg в нужной подпапке (Kids/Boys/Toys, Package/Ribbons и т.д.) и отдаёт ссылку на картинку.

Кратко порядок действий:

  1. Открыть таблицу Giftbox/Data base → Расширения → Apps Script.
  2. Вставить код ниже, заменить ROOT_FOLDER_ID на ID папки Giftbox (из её ссылки).
  3. Развернуть → Deploy → Web app → доступ «Все, у кого есть ссылка».
  4. Скопировать выданный URL (…/exec) и вставить его в поле выше.
// Вставьте в Code.gs таблицы Data base
const ROOT_FOLDER_ID = 'ВАШ_ID_ПАПКИ_Giftbox';

// doGet поддерживает обычный JSON-ответ и JSONP (?callback=...),
// потому что при открытии HTML-файла напрямую (file://) браузер
// не даёт читать обычный fetch-ответ Apps Script (ограничение
// самого браузера, а не таблицы) — JSONP это ограничение обходит.
function doGet(e){
  let payload;
  if(e.parameter.action === 'image'){
    payload = getImagePayload(e.parameter.path, e.parameter.id);
  } else {
    payload = readAll();
  }
  return output(payload, e.parameter.callback);
}

function doPost(e){
  const body = JSON.parse(e.postData.contents);
  if(body.action === 'save'){
    saveProductsAndStores(body.data);
  } else if(body.action === 'logOrder'){
    appendHistoryRow(body.record);
  }
  return output({ok:true});
}

function output(payload, callback){
  const json = JSON.stringify(payload);
  if(callback){
    return ContentService.createTextOutput(callback + '(' + json + ');')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function readAll(){
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  return {
    products: sheetToObjects(ss.getSheetByName('Products')),
    stores:   sheetToObjects(ss.getSheetByName('Stores')),
    history:  sheetToObjects(ss.getSheetByName('History'))
  };
}
function sheetToObjects(sheet){
  if(!sheet) return [];
  const [header, ...rows] = sheet.getDataRange().getValues();
  return rows.filter(r=>r.join('')!=='').map(r=>{
    const o={}; header.forEach((h,i)=>o[h]=r[i]); return o;
  });
}

const toCell = (r,h) => {
  const v = r[h];
  if(v===undefined || v===null) return '';
  return typeof v==='object' ? JSON.stringify(v) : v;
};

// ВАЖНО: товары НИКОГДА не удаляются из листа Products целиком.
// Когда пользователь нажимает «удалить» в приложении, строка остаётся
// на месте, только столбец qty (остаток) заменяется на "-" — это
// специально просимая логика: если в будущем понадобится вернуть товар,
// достаточно вручную вписать число в ячейку qty в самой таблице, и на
// следующей синхронизации товар снова появится в приложении.
// Магазины (Stores) удаляются по-настоящему — там нет остатка, который
// можно было бы "занулить", и явного требования хранить их архивно не было.
function saveProductsAndStores(data){
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const productsHeader = ['id','name','section','gender','category','qty','emoji'];
  upsertKeyed(ss, 'Products', data.products, productsHeader, 'id');
  upsertKeyed(ss, 'Stores', data.stores, ['name','addr','cat'], 'name');
  archiveByKey(ss, 'Products', productsHeader, 'id', 'qty', data.archivedProducts || []);
  deleteByKey(ss, 'Stores', ['name','addr','cat'], 'name', data.deletedStores || []);
}

// Обновляет/добавляет строки по ключу keyField, НИЧЕГО не удаляя сама:
// - существующая строка (совпал ключ) — обновляется на месте (остаток 0 в т.ч.)
// - новая строка — дописывается в конец
function upsertKeyed(ss, name, rows, header, keyField){
  let sh = ss.getSheetByName(name) || ss.insertSheet(name);
  if(sh.getLastRow()===0) sh.appendRow(header);

  const existingCount = Math.max(0, sh.getLastRow()-1);
  const keyIdx = header.indexOf(keyField);
  const keyToRow = {};
  if(existingCount>0){
    const existing = sh.getRange(2,1,existingCount,header.length).getValues();
    existing.forEach((r,i)=>{
      const k = String(r[keyIdx]);
      if(k!=='') keyToRow[k] = i+2;
    });
  }

  (rows||[]).forEach(r=>{
    const key = String(r[keyField]);
    const values = header.map(h=>toCell(r,h));
    if(keyToRow[key]){
      sh.getRange(keyToRow[key],1,1,header.length).setValues([values]); // обновляем на месте, включая qty=0
    } else {
      sh.appendRow(values); // новая строка — в конец
    }
  });
}

// Ставит "-" в столбец qtyField для строк с указанными ключами, СТРОКУ НЕ УДАЛЯЕТ.
// Это и есть "мягкое удаление" товара: остаток становится "-", а не пропадает.
function archiveByKey(ss, name, header, keyField, qtyField, keysToArchive){
  if(!keysToArchive || keysToArchive.length===0) return;
  const sh = ss.getSheetByName(name);
  if(!sh) return;
  const existingCount = Math.max(0, sh.getLastRow()-1);
  if(existingCount===0) return;
  const keyIdx = header.indexOf(keyField);
  const qtyIdx = header.indexOf(qtyField);
  const existing = sh.getRange(2,1,existingCount,header.length).getValues();
  const wanted = new Set(keysToArchive.map(String));
  existing.forEach((r,i)=>{
    if(wanted.has(String(r[keyIdx]))){
      sh.getRange(i+2, qtyIdx+1).setValue('-');
    }
  });
}

// Удаляет строки строго по явному списку ключей (используется только для Stores —
// у магазинов нет остатка, поэтому "архивировать через прочерк" для них не применимо)
function deleteByKey(ss, name, header, keyField, keysToDelete){
  if(!keysToDelete || keysToDelete.length===0) return;
  const sh = ss.getSheetByName(name);
  if(!sh) return;
  const existingCount = Math.max(0, sh.getLastRow()-1);
  if(existingCount===0) return;
  const keyIdx = header.indexOf(keyField);
  const existing = sh.getRange(2,1,existingCount,header.length).getValues();
  const wanted = new Set(keysToDelete.map(String));
  const rowsToDelete = [];
  existing.forEach((r,i)=>{
    if(wanted.has(String(r[keyIdx]))) rowsToDelete.push(i+2);
  });
  rowsToDelete.sort((a,b)=>b-a).forEach(rowNum=> sh.deleteRow(rowNum)); // снизу вверх, чтобы не сбить номера строк
}

// Дописывает ОДНУ строку заказа в History и НИКОГДА не трогает уже существующие
// строки и не пересылает/не перечитывает весь журнал целиком. Раньше вся история
// целиком пересобиралась при каждом save() — с ростом журнала это упиралось
// в лимиты времени выполнения Apps Script, и сохранение молча переставало работать.
// Теперь один вызов = одна новая строка, независимо от размера журнала.
//
// ВАЖНО: заголовок листа читается из САМОГО листа, а не считается фиксированным.
// Если лист History был создан ДО того, как в записи заказа появилось поле
// "gender" (старый формат — только order/date/items), то жёстко прописанный
// 4-колоночный заголовок "съезжал" бы данные на одну ячейку (gender попадал бы
// в столбец date, date — в столбец items, а сам items вообще терялся) — именно
// из-за этого при чтении история переставала парситься. Теперь недостающие
// столбцы дописываются в конец, не трогая уже существующие данные.
function appendHistoryRow(record){
  if(!record) return;
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const desiredFields = ['order','gender','date','ts','items'];
  let sh = ss.getSheetByName('History') || ss.insertSheet('History');
  if(sh.getLastRow()===0) sh.appendRow(desiredFields);

  let header = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
  desiredFields.forEach(f=>{
    if(header.indexOf(f) === -1){
      sh.getRange(1, header.length+1).setValue(f); // недостающее поле — новым столбцом в конец
      header = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
    }
  });

  const values = header.map(h => toCell(record, h)); // порядок значений строго по РЕАЛЬНОМУ заголовку листа
  sh.appendRow(values);
}

// path пример: "Items/Kids/Boys/Toys" или "Package/Ribbons"
// Сначала пробуем найти папку строго по пути (без учёта регистра/пробелов),
// а если не нашли (например, папка на Диске названа чуть иначе) —
// ищем файл с нужным ID по всему дереву папок Giftbox, чтобы картинка
// подтягивалась в любом случае (это чинит проблему с коробками).
function getImagePayload(path, id){
  const root = DriveApp.getFolderById(ROOT_FOLDER_ID);

  let folder = root;
  let pathOk = true;
  path.split('/').forEach(part=>{
    if(!pathOk) return;
    const found = findSubfolder(folder, part);
    if(found) folder = found; else pathOk = false;
  });
  if(pathOk){
    const hit = findFileInFolder(folder, id);
    if(hit) return hit;
  }

  const hit2 = findFileRecursive(root, id);
  if(hit2) return hit2;

  return { dataUrl: null };
}
function findSubfolder(folder, name){
  const target = name.trim().toLowerCase();
  const it = folder.getFolders();
  while(it.hasNext()){
    const f = it.next();
    if(f.getName().trim().toLowerCase() === target) return f;
  }
  return null;
}
function findFileInFolder(folder, id){
  const files = folder.getFiles();
  while(files.hasNext()){
    const f = files.next();
    const base = f.getName().replace(/\.[^.]+$/, '');
    if(base === id){
      const blob = f.getBlob();
      const b64 = Utilities.base64Encode(blob.getBytes());
      return { dataUrl: 'data:'+blob.getContentType()+';base64,'+b64 };
    }
  }
  return null;
}
function findFileRecursive(folder, id){
  const direct = findFileInFolder(folder, id);
  if(direct) return direct;
  const subs = folder.getFolders();
  while(subs.hasNext()){
    const hit = findFileRecursive(subs.next(), id);
    if(hit) return hit;
  }
  return null;
}

Таблица «Data base» должна содержать 3 листа: Products (id, name, section, gender, category, qty, emoji), Stores (name, addr, cat), History (order, gender, date, items). Картинки при этом остаются в папках как есть — переименовывать/перемещать их не нужно.

Важно: замените код на этот и передеплойте: Deploy → Manage deployments → ✎ → Version: New version → Deploy. Ссылка (…/exec) при этом не изменится. Это обязательно нужно сделать после обновления файла, иначе будет работать старая логика скрипта.

Что изменилось в этой версии скрипта: раньше строки Products/Stores удалялись из таблицы, если товара не было в присланных данных — из-за этого товар с остатком 0 иногда мог целиком пропасть из таблицы. Теперь остаток 0 просто пишется как 0 и строка остаётся. При удалении товара в приложении (кнопка «удалить» на вкладке «Склад») строка в таблице тоже не удаляется — вместо неё в столбце qty ставится «-», товар пропадает из приложения, но если в будущем понадобится — достаточно вручную вписать число в эту ячейку в таблице, и товар снова появится в приложении сам собой при следующей синхронизации. Магазины по-прежнему удаляются из таблицы по-настоящему (для них нет «остатка», который можно было бы занулить). Также перед удалением приложение теперь спрашивает подтверждение («Да» / «Нет»). Кроме того, History теперь дописывается по одной строке за раз, а не пересылается целиком при каждом сохранении — это должно починить прекратившееся сохранение журнала заказов по мере его роста.

Если на телефоне не работают кнопки/вкладки

Чаще всего причина не в самой странице, а в том, что телефон открывает HTML-файл через встроенный просмотрщик (Google Диск, почта, мессенджер) — там JavaScript либо не выполняется, либо выполняется в урезанном режиме, поэтому вкладки не переключаются и не вставить текст в поле. Сама страница при этом исправна. Теперь страница сама пытается это заметить и показывает красную плашку сверху с кнопкой «Открыть в браузере» независимо от того, на какой вкладке вы застряли — раньше эта инструкция была только здесь, на вкладке «Подключение», куда как раз и нельзя было попасть при сломанной навигации.

  1. В приложении Google Диск на файле нажать «⋮» → Открыть в приложении → выбрать браузер (Chrome/Safari), а не оставлять открытым во встроенном просмотрщике Диска.
  2. Либо загрузить этот HTML-файл на бесплатный хостинг (например GitHub Pages, Netlify Drop или Google Sites) и открывать по обычной ссылке https:// — тогда кнопки, вкладки и вставка текста будут работать как на компьютере, даже если ссылку прислали в мессенджере.
